#### 实验任务

创建 `student_score.py` 文件。手工编写并运行以下代码，模拟一个完整的“地理信息系统（GIS）专业学生成绩录入与统计系统”：

Python

```
# -*- coding: utf-8 -*-
"""
程序名称：student_score.py
作者：黄乐源
日期：2026-07-05
描述：掌握列表（List）的定义、高频方法（append/pop等）、切片操作，实现成绩动态录入与空间统计
"""

print("================= 任务 1：列表基础操作与动态管理 =================")
# 初始化一个 GIS 专业核心学生名单列表
gis_students = ["张三", "李四", "王五"]
print(f"原始学生列表: {gis_students}")

# 1. append() 追加新学员
gis_students.append("赵六")
print(f"追加赵六后: {gis_students}")

# 2. insert() 在指定索引插入学员（模拟插班生）
gis_students.insert(1, "钱七")
print(f"在索引1位置插入钱七后: {gis_students}")

# 3. extend() 批量合并另一个科研小组的名单
new_group = ["周八", "吴九"]
gis_students.extend(new_group)
print(f"合并新小组后: {gis_students}")

# 4. index() 查找特定学生在列表中的索引位置
search_name = "王五"
if search_name in gis_students:
    idx = gis_students.index(search_name)
    print(f"🔍 查询成功！{search_name} 在列表中的位置索引为: {idx}")


print("\n================= 任务 2：高频切片演练（空间数据截取模拟） =================")
# 模拟对一个长序列空间栅格像元或学生列表进行切片
# 提取索引 1 到 5（不含5）的学员名单
sub_list1 = gis_students[1:5]
print(f"切片 [1:5] 截取段: {sub_list1}")

# 步长切片：隔一个取一个（模拟遥感影像降采样空间抽样）
sub_list2 = gis_students[::2]
print(f"步长为2的抽样截取 [::2]: {sub_list2}")

# 逆序切片：利用负步长反转列表
reverse_list = gis_students[::-1]
print(f"利用切片实现列表完全反转 [::-1]: {reverse_list}")


print("\n================= 任务 3：综合实战（华师大资环学院成绩统计系统） =================")
scores = [] # 初始化空列表，用于动态存储成绩
print("📊 欢迎进入 Python 基本功测试：学生成绩统计中心")
print("💡 提示：请输入学生成绩（0-100的数字），输入 -1 结束录入。")

while True:
    input_val = input("请输入学生分数: ").strip()
    
    # 转换为浮点数并检查退出条件
    score = float(input_val)
    if score == -1:
        break
        
    if 0 <= score <= 100:
        scores.append(score) # 动态追加到列表末尾
        print(f"已录入分数: {score}，当前总录入人数: {len(scores)}")
    else:
        print("⚠️ 警告：分数范围必须在 0 到 100 之间，请重新输入！")

# 统计与计算模块
if len(scores) > 0:
    total_score = sum(scores)      # 内置函数求和
    student_count = len(scores)    # 内置函数求长度
    average_score = total_score / student_count # 计算平均分
    
    print("\n--- 📈 华师大资环学院统计结果汇报 ---")
    print(f"总计录入人数: {student_count} 人")
    print(f"最高分数: {max(scores)} 分")
    print(f"最低分数: {min(scores)} 分")
    print(f"班级总成绩: {total_score:.2f} 分")
    print(f"💡 班级平均分: {average_score:.2f} 分")
else:
    print("❌ 录入中心未检测到有效的分数数据，无法完成统计。")

print("\n🏁 Day08 列表核心实验全部顺利达成！")
```
