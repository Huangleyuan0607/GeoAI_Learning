#### 实验任务

创建 `student_management.py` 文件。手工编写并运行以下代码，前半部分构建基于面向对象的空间要素增删改查管理系统，后半部分加入严谨的 `try...except...else...finally` 异常捕获机制，模拟脏数据输入的防御审计：  

```python
# -*- coding: utf-8 -*-
"""
程序名称：student_management.py
作者：黄乐源
日期：2026-07-12
描述：面向对象空间要素管理系统与异常处理机制实战，完成 Python 基础阶段收官交付
"""

print("================= 任务 1：面向对象空间要素(Building)增删改查管理系统 =================")

class GeoFeature:
    """地理空间要素数据实体类"""
    def __init__(self, feature_id: str, feature_type: str, area: float):
        self.feature_id: str = feature_id
        self.feature_type: str = feature_type
        self.area: float = area

    def __str__(self) -> str:
        return f"📌 [要素 ID: {self.feature_id:<12} | 类型: {self.feature_type:<10} | 面积: {self.area:.2f} m²]"

class GeoFeatureManager:
    """空间要素业务管理类：负责 CRUD 维护"""
    def __init__(self):
        self.feature_list: list[GeoFeature] = []  # 存储 GeoFeature 对象的列表

    def add_feature(self, feature: GeoFeature) -> None:
        """添加要素"""
        self.feature_list.append(feature)
        print(f"✅ 成功添加要素: {feature.feature_id}")

    def query_all(self) -> None:
        """查询打印全部要素"""
        print("\n🌐 --- 华中师范大学城环学院 GeoAI 要素数据库清单 ---")
        if not self.feature_list:
            print("⚠️ 数据库当前为空！")
            return
        for item in self.feature_list:
            print(item)

# 实例化管理器并录入初始数据
manager = GeoFeatureManager()
manager.add_feature(GeoFeature("BLD_001", "building", 1520.5))
manager.add_feature(GeoFeature("WAT_002", "water", 8900.0))
manager.query_all()


print("\n================= 任务 2：脏数据输入与异常处理(try...except...else...finally)防御 =================")

def parse_remote_sensing_scale(raw_input: str) -> float:
    """
    解析用户输入的遥感分辨率比例尺，演示严谨的异常处理装甲
    """
    print(f"\n📡 正在尝试解析原始分辨率参数字符串: '{raw_input}'")
    try:
        # 尝试转换为浮点数，若输入非数字字符（如 "invalid"）会引发 ValueError
        scale = float(raw_input)
        
        # 若数值非法，手动触发 ZeroDivisionError 或 ValueError
        if scale <= 0:
            raise ValueError("分辨率比例尺必须严格大于 0！")
            
        calculated_resolution = 1.0 / scale  # 正常计算

    except ValueError as e:
        # 捕获数值转换错误或手动抛出的异常
        print(f"❌ [捕获异常] 转换失败或参数越界: {e}")
        return -1.0
    except ZeroDivisionError:
        # 捕获除零异常
        print("❌ [捕获异常] 比例尺不能为 0，无法计算分辨率！")
        return -1.0
    except Exception as e:
        # 兜底捕获未知的其他所有异常
        print(f"❌ [未知异常] 发生未知运行时错误: {e}")
        return -1.0
    else:
        # 只有在 try 块代码完全正常、没有发生任何异常时才会执行
        print(f"🎉 [解析成功] 算子运行正常，计算得出地面实际分辨率: {calculated_resolution:.4f} 米/像素")
        return calculated_resolution
    finally:
        # 无论是否发生异常，都一定会执行的资源清理/日志收尾动作
        print("🔒 [系统日志] 本次分辨率参数解析审计完毕，释放通道资源。")

# 测试 1：正常输入合法数据
res1 = parse_remote_sensing_scale("0.5")

# 测试 2：输入包含非法字母的脏数据
res2 = parse_remote_sensing_scale("invalid_30m")

# 测试 3：输入非法的零值
res3 = parse_remote_sensing_scale("0")

print("\n🏁 Day15 综合要素管理与异常防爆系统测试完毕，第一阶段 Python 基础圆满收官！")
```
